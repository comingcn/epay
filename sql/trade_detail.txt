CREATE TABLE `data_analysis`.`trade_detail` (
  `ID` INT NOT NULL AUTO_INCREMENT COMMENT '记录id,自增长',
  `TXN_SEQ_ID` INT NOT NULL COMMENT '交易序列号',
  `TXN_DATE` TIMESTAMP NOT NULL COMMENT '交易日期',
  `CERT_NO` VARCHAR(17) NOT NULL COMMENT '身份证号',
  `CARD_NO` VARCHAR(32) NOT NULL COMMENT '银行卡号',
  `MER_ID` VARCHAR(15) NOT NULL COMMENT '商户ID',
  `MER_TYPE` VARCHAR(2) NOT NULL COMMENT '商户类型',
  `AMOUNT` DECIMAL NOT NULL COMMENT '交易金额（单位元）',
  `SF_TYPE` CHAR NOT NULL COMMENT '收付类型',
  `RETURN_CODE` VARCHAR(4) NOT NULL COMMENT '交易状态码',
  PRIMARY KEY (`ID`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
COMMENT = '交易明细表';
